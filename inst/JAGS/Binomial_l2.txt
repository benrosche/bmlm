model {
  
  # Likelihood
  for(j in 1:n.l2) {
    # Logistic regression
    Y[j] ~ dbern(p[j])
    # Link function
    logit(p[j]) <- l2[j]
    # Level 2: systematic component
    l2[j] <- inprod(X.l2[j,], b.l2)
  }
  
  # Prior: regression coefficients
  for(x in 1:n.Xl2) {
    b.l2[x] ~ dnorm(0,0.0001)
  }
  
  # Variance of logistic distribution 
  sigma.l2 <- 3.14159^2/3 
  
}
