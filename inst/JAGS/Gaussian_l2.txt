model {
  
  # Likelihood
  for(j in 1:n.l2) {
    # Linear regression
    Y[j] ~ dnorm(mu[j], tau.l2)
    # Link function
    mu[j] <- l2[j] 
    # Level 2: systematic component
    l2[j] <- inprod(X.l2[j,], b.l2)
  }
  
  # Prior: regression coefficients
  for(x in 1:n.Xl2) {
    b.l2[x] ~ dnorm(0,0.0001)
  }

  # Prior: variance terms 
  tau.l2 ~ dscaled.gamma(25, 1)
  sigma.l2 <- 1/sqrt(tau.l2)
  
}
